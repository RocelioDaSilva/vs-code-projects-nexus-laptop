═══════════════════════════════════════════════════════════════════════════════════

1️⃣  PERGUNTAS DE SCREENING TÉCNICO (Tabela 4 Artigo)
    ├─ Injeção de Vapor: 6 perguntas críticas
    ├─ Combustão In Situ: 6 perguntas
    ├─ CO₂ Miscível: 6 perguntas
    ├─ Gás Imiscível: 6 perguntas
    ├─ Polímeros: 6 perguntas
    ├─ Surfactantes: 6 perguntas
    ├─ Alcalina: 6 perguntas
    ├─ Água Inteligente: 6 perguntas
    ├─ Térmico Deepwater: 6 perguntas
    └─ Microbiano: [Variável]

2️⃣  CLASSIFICAÇÃO OFFSHORE (SPE IADC)
    ├─ Águas Rasas: 0-500m subsea (1.0x custo)
    ├─ Águas Intermediárias: 500-1500m (2.5x custo)
    ├─ Águas Profundas: 1500-3000m (5.0x custo)
    └─ Águas Ultraprofundas: 3000-6000m (8.0x custo)

3️⃣  CAMPOS DE REFERÊNCIA ANGOLA
    ├─ Bloco 15: Onshore/Transição (50m água)
    ├─ Bloco 17: Offshore Intermediário (800m água)
    ├─ Bloco 18: Offshore Profundo (1200m água)
    ├─ Bloco 31: Offshore Raso (300m água)
    └─ Cabinda: Onshore/Transição (100m água)

4️⃣  VALIDAÇÃO DE CONSISTÊNCIA (7 critérios)
    ├─ Saturação total (So + Sw ≤ 100%)
    ├─ API vs Viscosidade
    ├─ Profundidade vs Temperatura (gradiente 1°C/30m)
    ├─ Profundidade vs Pressão (gradiente 0.45 psi/m)
    ├─ Porosidade vs Permeabilidade
    ├─ TAN mínimo para química (> 0.1)
    └─ pH vs Salinidade

5️⃣  CÁLCULO DE EFICIÊNCIA
    ├─ Número Capilar (Nc = v*μ / σ*cosθ)
    ├─ Eficiência Microscópica (PSD)
    ├─ Eficiência Macroscópica (SE)
    └─ Fator de Recuperação (RF = PSD × SE × D × T)

6️⃣  REGRAS OFFSHORE POR PROFUNDIDADE
    ├─ Métodos Térmicos: Inviáveis > 2000m subsea
    ├─ Métodos Miscíveis: Requerem MMP adequado
    ├─ Métodos Químicos: Risco degradação > 3000m
    └─ Todos: Espaçamento poços cresce com profundidade

💡 EXEMPLOS DE USO RÁPIDO
═══════════════════════════════════════════════════════════════════════════════════

# 1. OBTER PERGUNTAS DE SCREENING
─────────────────────────────────────
from v7 import AdvancedScreeningQuestions

questions = AdvancedScreeningQuestions.get_questions_by_method("Injeção de Vapor")
category = AdvancedScreeningQuestions.get_category("Injeção de Vapor")

print(f"Categoria: {category}")
for i, q in enumerate(questions, 1):
    print(f"{i}. {q}")


# 2. VALIDAR CONSISTÊNCIA DE DADOS
─────────────────────────────────────
from v7 import DataValidator

campo = {
    'API': 18.5,
    'Viscosidade': 850,
    'Profundidade': 1200,
    'Temperatura': 55,
    'Saturação de Óleo': 62,
    'Saturação de Água': 35
}

valid, messages = DataValidator.validate_consistency(campo)

if valid:
    print("✓ Dados consistentes")
else:
    for msg in messages:
        print(f"⚠ {msg}")


# 3. ANALISAR VIABILIDADE OFFSHORE
─────────────────────────────────────
from v7 import OffshoreSpecificCriteria

viable, reason = OffshoreSpecificCriteria.validate_offshore_feasibility(
    method_name="Injeção de Vapor",
    depth_subsea=800,
    reservoir_depth=2800
)

classification = OffshoreSpecificCriteria.get_water_depth_classification(800)
cost_mult = OffshoreSpecificCriteria.get_cost_multiplier(800)

print(f"Classificação: {classification}")
print(f"Multiplicador de custo: {cost_mult}x")
print(f"Viável: {'✓' if viable else '✗'} - {reason}")


# 4. CALCULAR EFICIÊNCIA (NÚMERO CAPILAR)
─────────────────────────────────────────
from v7 import EfficiencyCalculator

nc = EfficiencyCalculator.calculate_capillary_number(
    velocity=1.0,           # ft/dia
    viscosity=100,          # cP
    ift=25,                 # dinas/cm
    contact_angle=30        # graus
)

interpretation = EfficiencyCalculator.interpret_capillary_number(nc)
print(f"Número Capilar: {nc:.2e}")
print(f"Nível: {interpretation['level']}")
print(f"Potencial: {interpretation['recovery_potential']}")


# 5. CALCULAR FATOR DE RECUPERAÇÃO
─────────────────────────────────────
from v7 import EfficiencyCalculator

rf = EfficiencyCalculator.calculate_recovery_factor(
    psd=0.85,               # Eficiência microscópica
    se=0.70,                # Eficiência macroscópica
    drainage=0.95,          # Drenagem
    time_factor=0.90        # Fator temporal
)

print(f"RF Total: {rf['RF_percentage']:.1f}%")
print(f"  PSD: {rf['PSD']*100:.0f}%")
print(f"  SE: {rf['SE']*100:.0f}%")
print(f"  D: {rf['D']*100:.0f}%")
print(f"  T: {rf['T']:.2f}x")
