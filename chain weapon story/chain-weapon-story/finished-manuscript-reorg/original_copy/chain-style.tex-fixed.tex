% --------------------
% Chapter stripe with per-chapter markers (like Bible margin stripes)
% Usage: \\SetTotalChapters{<n>} to enable marks. If n==0 the stripe is hidden.
% Place this in the preamble to set the total chapter count for the book.
\\newcounter{CWTotalChapters}
\\newcommand{\\SetTotalChapters}[1]{\\setcounter{CWTotalChapters}{#1}}
\\SetTotalChapters{0}

\\AddEverypageHook{%
  \\ifnum\\value{CWTotalChapters}>0
    \\ifnum\\value{chapter}>0
      \\begin{tikzpicture}[remember picture,overlay]
        % choose outer side depending on page parity
        \\ifodd\\value{page}
          \\coordinate (sttop) at (current page text area.north east);
          \\coordinate (stbot) at (current page text area.south east);
          \\def\\xshift{6mm}
        \\else
          \\coordinate (sttop) at (current page text area.north west);
          \\coordinate (stbot) at (current page text area.south west);
          \\def\\xshift{-6mm}
        \\fi
        % stripe shaft
        \\fill[CWAccent!65] ($(sttop)+(\\xshift,0)$) rectangle ($(stbot)+(\\xshift-4mm,0)$);
        % chapter markers along the text area height
        \\pgfmathtruncatemacro{\\nchap}{\\value{CWTotalChapters}}
        \\foreach \\i in {1,...,\\nchap} {%
          \\pgfmathsetmacro{\\t}{ifthenelse(\\nchap>1, (\\i-1)/(\\nchap-1), 0.5)}%
          \\coordinate (mark) at ($(sttop)!\\t!(stbot)$);
          \\coordinate (markout) at ($(mark)+(\\xshift-2mm,0)$);
          \\ifnum\\i=\\value{chapter}
            \\draw[fill=white, draw=CWAccent, line width=0.9pt] (markout) circle (1.6mm);
          \\else
            \\draw[fill=white!85, draw=CWAccent!30] (markout) circle (0.9mm);
          \\fi
        }
        % small chapter number rotated in the stripe
        \\node[rotate=90, font=\\sffamily\\bfseries, text=white] at ($(sttop)!0.5!(stbot)+(\\xshift-9mm,0)$) {\\thechapter};
      \\end{tikzpicture}%
    \\fi
  \\fi
}
