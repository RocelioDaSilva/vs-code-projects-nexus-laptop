% Generic shape constructor: Cavern (smooth flowing stone layers)
\newcommand{\LandscapeShapeCavern}[3]{%
  % #1 = sky, #2 = wall, #3 = depth
  \fill[#1] (TL) rectangle ($(TR)-(0,\SH cm)$);
  \fill[#2!70]
    ($(TL)-(0,\SH cm*.20)$) 
    .. controls ($(TL)!.25!(TR)-(0,.08cm)$) and ($(TL)!.75!(TR)-(0,.08cm)$) ..
    ($(TR)-(0,\SH cm*.20)$) -- ($(TR)-(0,\SH cm)$) -- ($(TL)-(0,\SH cm)$) -- cycle;
  \fill[#3]
    ($(TL)-(0,\SH cm*.40)$)
    .. controls ($(TL)!.20!(TR)+(0,.10cm)$) and ($(TL)!.80!(TR)+(0,.10cm)$) ..
    ($(TR)-(0,\SH cm*.40)$) -- ($(TR)-(0,\SH cm)$) -- ($(TL)-(0,\SH cm)$) -- cycle;
}

}

% Generic shape: Jungle (dense, layered vegetation)

  % #1 = sky, #2 = foliage, #3 = depth
  \fill[#1] (TL) rectangle ($(TR)-(0,\SH cm)$);
  \fill[#2!80]
    ($(TL)-(0,\SH cm*.25)$) 
    -- ($(TL)!.10!(TR)-(0,\SH cm*.15)$) -- ($(TL)!.25!(TR)-(0,\SH cm*.40)$)
    -- ($(TL)!.40!(TR)-(0,\SH cm*.20)$) -- ($(TL)!.55!(TR)-(0,\SH cm*.45)$)
    -- ($(TL)!.70!(TR)-(0,\SH cm*.18)$) -- ($(TL)!.85!(TR)-(0,\SH cm*.38)$)
    -- ($(TR)-(0,\SH cm*.25)$) -- ($(TR)-(0,\SH cm)$) -- ($(TL)-(0,\SH cm)$) -- cycle;
  \fill[#3]
    ($(TL)-(0,\SH cm*.50)$)
    -- ($(TL)!.15!(TR)-(0,\SH cm*.48)$) -- ($(TL)!.30!(TR)-(0,\SH cm*.52)$)
    -- ($(TL)!.50!(TR)-(0,\SH cm*.48)$) -- ($(TL)!.70!(TR)-(0,\SH cm*.52)$)
    -- ($(TL)!.85!(TR)-(0,\SH cm*.48)$) -- ($(TR)-(0,\SH cm*.50)$)
