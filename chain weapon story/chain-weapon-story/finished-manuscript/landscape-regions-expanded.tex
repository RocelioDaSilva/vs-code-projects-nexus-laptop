\newcommand{\LandscapeBodyDesertBase}[3]{%
  % #1 = background colour, #2 = dune colour, #3 = shadow colour
  \fill[#1] (TL) rectangle ($(TR)-(0,\SH cm)$);
  \fill[#2!70]
    ($(TL)-(0,\SH cm*.25)$) 
    .. controls ($(TL)!.10!(TR)+(0,.08cm)$) and ($(TL)!.20!(TR)-(0,.08cm)$) ..
    ($(TL)!.35!(TR)-(0,\SH cm*.35)$)
    .. controls ($(TL)!.50!(TR)+(0,.10cm)$) and ($(TL)!.65!(TR)-(0,.12cm)$) ..
    ($(TL)!.80!(TR)-(0,\SH cm*.28)$)
    .. controls ($(TL)!.90!(TR)+(0,.08cm)$) ..
    ($(TR)-(0,\SH cm*.20)$) -- ($(TR)-(0,\SH cm)$) -- ($(TL)-(0,\SH cm)$)--cycle;
  \fill[#3]
    ($(TL)-(0,\SH cm*.40)$)
    .. controls ($(TL)!.15!(TR)+(0,.06cm)$) and ($(TL)!.30!(TR)-(0,.10cm)$) ..
    ($(TL)!.45!(TR)-(0,\SH cm*.50)$)
    .. controls ($(TL)!.60!(TR)+(0,.12cm)$) and ($(TL)!.75!(TR)-(0,.08cm)$) ..
    ($(TL)!.92!(TR)-(0,\SH cm*.40)$) -- ($(TR)-(0,\SH cm)$) -- ($(TL)-(0,\SH cm)$)--cycle;
}
